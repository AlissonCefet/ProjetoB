CREATE TABLE USUARIOS (
   NOME VARCHAR(100),
   SENHA VARCHAR(32),
   EMAIL VARCHAR(100),
   ENDERECO VARCHAR(100),
   NUMERO VARCHAR(10),
   ESTADO VARCHAR(2),
   CIDADE VARCHAR(50),
   TELEFONE VARCHAR(14),
   CELULAR VARCHAR(14),
   CPF VARCHAR(14),
   DATA_NASCIMENTO VARCHAR(10),
   SEXO VARCHAR(1)
);


CREATE TABLE PRODUTOS (
CODIGO INTEGER NOT NULL PRIMARY KEY,
NOME VARCHAR(100),
MARCA VARCHAR(100),
UNIDADE VARCHAR(15),
PRECO DOUBLE PRECISION,
TIPO VARCHAR(30),
QUANTIDADE DOUBLE PRECISION
);
